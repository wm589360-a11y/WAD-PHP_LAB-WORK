5. Write a PHP program to print 5 to 10 using For and For Each.

<?php

// for loop
for ($i = 5; $i <= 10; $i++) {
    echo $i . "<br>";
}
?>
-------------------------------------------------------------------------------------------------------------
PROGRAM OUTPUT:

5
6
7
8
9
10
-------------------------------------------------------------------------------------------------------------
<?php
// Foreach loop

$numbers = array(5, 6, 7, 8, 9, 10);
foreach ($numbers as $number) {
    echo $number . "<br>";
}
?>
-------------------------------------------------------------------------------------------------------------
PROGRAM OUTPUT:

5
6
7
8
9
10
